#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int n;
int s[10001];
int main() {
    int c = 0;
    scanf("%d",&n);
    for(int i = 1;i <= n;i ++) {
        scanf("%d",&s[i]);
    }
    for(int i = 1;i <= n;i ++) {
        while(s[i] != i) {
            c ++;
            swap(s[i],s[s[i]]);
        }
    }
    printf("%d",c);
}
