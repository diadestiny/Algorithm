```cpp
#include<iostream>

using namespace std;
int a,b,c;
int n[100010],m[100010];
int main(){
    cin>>a>>b>>c;
    for(int i=0;i<a;i++)cin>>n[i];
    for(int j=0;j<b;j++) cin>>m[j];
    for(int i=0,j=b-1;i<a;i++){
        while(n[i]+m[j]>c) j--;
        if(n[i]+m[j] == c) {
            cout<<i<<" "<<j<<endl;
            break;
        }
    }
    
    return 0;
}
```
