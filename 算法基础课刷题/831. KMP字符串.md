```
#include<iostream>

using namespace std;
const int N=100100,M=1000100;
int n,m;
char p[N],s[M];
int ne[N];
int main(){
    cin>>n>> p+1 >>m>> s+1;
    //求next过程
    for(int i=2,j=0;i<=n;i++){
        while(j&&p[i]!=p[j+1]) j=ne[j];
        if(p[i]==p[j+1]) j++;
        ne[i] = j;
    }
    //kmp匹配过程
    for(int i=1,j=0;i<=m;i++){
        //j尽可能退<==>模板串p尽可能少移动
        while(j && s[i]!=p[j+1]) j = ne[j];
        if(s[i]==p[j+1]) j++;
        if(j == n){
            //匹配成功
            printf("%d ",i-n);
            j = ne[j];
        }
    }
    return 0;
}
```
