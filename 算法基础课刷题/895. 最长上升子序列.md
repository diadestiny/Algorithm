```
#include<bits/stdc++.h>
using namespace std;
int a[1010];
int f[1010];
int main(){
    int n;
    int res = 0;
    cin>>n;
    for(int i=1;i<=n;i++) cin>>a[i];
    for(int i=1;i<=n;i++){
        f[i] = 1;//只有a[i]一个数
        for(int j=1;j<i;j++){
            if(a[i]>a[j])  f[i]= max(f[i],f[j]+1);
        }
        res = max(res,f[i]);
    }
    cout<<res;
    return 0;
}
```
