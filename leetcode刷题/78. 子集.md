class Solution {
public:
    vector<vector<int>> ans;
    vector<int> path; 
    vector<int> flag;
    vector<vector<int>> subsets(vector<int>& nums) {
        flag = vector<int>(nums.size());
        dfs(nums,0);
        return ans;
    }

    void dfs(vector<int> & nums,int u){
        if(u==nums.size()) {
            ans.push_back(path);
            return ;
        }
        dfs(nums,u+1);
        if(!flag[u]){
            flag[u] = 1;
            path.push_back(nums[u]);
            dfs(nums,u+1);
            flag[u] = 0;
            path.pop_back();
        }

    }
};

class Solution {
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        vector<vector<int>> res;
        int n = nums.size();
        for (int i = 0; i < (1 << n); i ++ )
        {
            vector<int> temp;
            for (int j = 0; j < n; j ++ )
                if (i >> j & 1)
                    temp.push_back(nums[j]);
            res.push_back(temp);
        }
        return res;
    }
};
