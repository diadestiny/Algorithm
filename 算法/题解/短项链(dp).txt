#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
//dp
ll dp[3000][2000];
const ll MOD=998244353;
int main()
{
    int n,kk;
        cin>>n>>kk;
        ll ans=0;
        for(int fst=1;fst<=kk;fst++)
        {

            memset(dp,0,sizeof(dp));
            dp[1][fst]=1;
            for(int j=2;j<=n;j++)
            {
                for(int k=1;k<=kk;k++)
                {
                    if(k==1)
                     dp[j][k]=(dp[j-1][k]+dp[j-1][k+1])%MOD;
                     else if(k==kk)
                        dp[j][k]=(dp[j-1][k-1]+dp[j-1][k])%MOD;
                     else
                        dp[j][k]=(dp[j-1][k-1]+dp[j-1][k]+dp[j-1][k+1])%MOD;
                    // cout<<dp[j][k]<<" ";
                }
                //cout<<endl;
            }
            ll temp=0;
            for(int i=fst-1;i<=kk&&i<=fst+1;i++)
            temp+=dp[n][i];
            ans+=temp;
            ans%=MOD;
            //cout<<temp<<endl;
        }
        cout<<ans%MOD<<endl;
    return 0;
}
