#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    //freopen("3.in","r",stdin);
    int n;
    cin>>n;
    vector<int> num1(n);
    vector<int> num2(n);
    for(int i=0;i<n;i++) num1[i]=i+1;
    for(int i=0;i<n;i++) cin>>num2[i];
    int ans=1;
    do{
        if(num1==num2) break;
        ans++;
    }while(next_permutation(num1.begin(),num1.end()));
    cout<<ans<<endl;
    return 0;
}
