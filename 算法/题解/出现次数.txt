#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int num[10000007];
int main(){
   // freopen("3.in","r",stdin);
    int n;
    cin>>n;
    for(int i=0;i<n;i++) cin>>num[i];
    int tol=1,x=num[0];
    for(int i=1;i<n;i++) {
        if(num[i]==x) tol++;
        else tol--;
        if(tol<0) x=num[i],tol=1;
    }
    tol=0;
    for(int i=0;i<n;i++){
        if(num[i]==x) tol++;
    }
    if(tol>(n/2)) cout<<"Yes\n";
    else cout<<"No\n";
    return 0;
}
