#include<bits/stdc++.h>
using namespace std;
int main()
{
    int i,j,n,dp[1000];

    while(cin>>n){
            memset(dp,0,sizeof(dp));
     dp[0]=0;

      for(i=1;i<=n;i++){
       if(i-1>=0 && i-3<0 && i-5<0)
       dp[i]=dp[i-1]+1;

      if(i-1>=0 && i-3>=0 && i-5<0){
        dp[i]=min(dp[i-1],dp[i-3])+1;
      }
      if(i-1>=0 && i-3>=0 && i-5>=0){
        dp[i]=min(min(dp[i-1],dp[i-3]),dp[i-5])+1;
      }
      }
      cout<<n<<" ans:  "<<dp[n]<<endl;
    }
	return 0;
}
