
#include <iostream>

using namespace std;

#define _for(i,a,b) for(int i=(a);i<(b);i++)

typedef long long ll;

const int maxn=1e6+5,N=1e5+5;

int Tree[maxn],a[N],b[N],n;

ll ans,s[N];



inline int lowbit(int x){

    return (x&-x);

}



void add(int x,int val){

    for(int i=x;i<=maxn;i+=lowbit(i))

        Tree[i]+=val;

}



int get(int x){

    int sum=0;

    for(int i=x;i;i-=lowbit(i))

        sum+=Tree[i];

    return sum;

}



int main(){

    _for(i,1,N) s[i]=s[i-1]+i;

    scanf("%d",&n);

    _for(i,0,n){//Ç°ÄæÐò

        scanf("%d",&a[i]);

        add(a[i]+1,1);

        b[i] = i+1-get(a[i]+1);

    }

    memset(Tree,0,sizeof(Tree));

    for(int i=n-1;i>=0;i--){//ºóÄæÐò

        add(a[i]+1,1);

        b[i]+=get(a[i]);

    }

    _for(i,0,n)

        ans+=s[b[i]];

    printf("%lld",ans);

}
